import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner s = new Scanner(System.in);

        Map<Integer, String> map = new HashMap<>();

        // Read 4 space-separated inputs
        for (int i = 1; i <= 4; i++) {
            map.put(i, s.next()); // Read each word (space-separated)
        }

        int keyToRemove = s.nextInt(); // Read key to remove

        System.out.println("Mappings of HashMap are : " + map);

        map.remove(keyToRemove); // Remove the key from map

        System.out.println("Mappings after removal are : " + map);
    }
}
