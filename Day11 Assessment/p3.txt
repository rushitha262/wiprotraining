SELECT 
    e.FirstName,
    e.LastName,
    d.DepartmentName,
    e.Salary
FROM 
    Employees e
JOIN 
    Departments d ON e.DepartmentID = d.DepartmentID
WHERE 
    e.Salary > (
        SELECT AVG(Salary)
        FROM Employees e2
        WHERE e2.DepartmentID = e.DepartmentID
    )
ORDER BY 
    e.Salary ASC;
